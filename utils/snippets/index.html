Cheat-sheet pratiche d'esame

  • Niente onclick inline. Tutti i listener si registrano con addEventListener. renderCard() e mostraDettaglio() ricevono callback (onClick, onBottone, onIndietro) che catturano l'oggetto via closure — nessun find(p => p.id === id) dopo aver perso l'oggetto.
  • Schema-agnostico. Passa opts.campi = { titolo, testo, immagine, prezzo, categoria } a renderCard/templateModal/mostraDettaglio — funzionano con qualunque schema (nome/name, descrizione/description, ecc.).
  • Filtri categoria: consigliato inizializzaFiltriBottoni (bottoni con stato .attivo). Alternative: inizializzaDropdown (tendina UL/LI) o <select> nativo come fallback rapido.
  • Event delegation dove si fa re-render: il listener sta sul contenitore (delega(...)), non sui figli.
  • Guard di init in ogni pagina: initSeEsiste('.griglia', () => ...)
  • Debounce sulla ricerca testuale: debounce(fn, 200) da filtri.js.

Boilerplate JS di catalogo (con gli utils)

// Prima di </body>, in ordine — niente defer:
//   <script src="utils/js/dom.js"></script>
//   <script src="utils/js/filtri.js"></script>
//   <script src="utils/js/dropdown.js"></script>
//   <script src="utils/js/componenti.js"></script>
//   <script src="utils/js/logging.js"></script>
//   <script src="script.js"></script>

const piante = [ /* ... */ ];
let categoriaCorrente = 'tutte';
let ricercaCorrente   = '';

const CAMPI = { titolo:'nome', testo:'descrizione', immagine:'immagine',
                prezzo:'prezzo', categoria:'categoria' };

function renderCatalogo() {
    const griglia = $('.griglia');
    griglia.innerHTML = '';
    const risultati = filtraSemplice(piante, categoriaCorrente, ricercaCorrente,
                                     ['nome', 'descrizione']);
    risultati.forEach(p => {
        griglia.appendChild(renderCard(p, {
            campi:    CAMPI,
            esaurito: x => !x.disponibile,
            onClick:  pianta => apriDettaglio(pianta)
        }));
    });
    if (risultati.length === 0) {
        griglia.innerHTML = '<p>Nessun risultato.</p>';
    }
    logFiltro(categoriaCorrente, ricercaCorrente, risultati);
}

function apriDettaglio(pianta) {
    logDettaglio(pianta);
    apriModal('.modal-overlay', templateModal(pianta, { ...CAMPI, extra:'cura' }));
}

document.addEventListener('DOMContentLoaded', () => {
    logInit('piante', piante);

    initSeEsiste('.griglia', () => {
        // Filtri a bottoni (consigliato). Se preferisci la tendina:
        // inizializzaDropdown('[data-dropdown="categoria"]', ...)
        inizializzaFiltriBottoni('[data-filtri="categoria"]',
            estraiCategorie(piante, 'categoria'),
            cat => { categoriaCorrente = cat; renderCatalogo(); },
            { valoreIniziale:'tutte' });

        const cerca = debounce(v => { ricercaCorrente = v; renderCatalogo(); }, 200);
        ascolta('#input-ricerca', 'input', e => cerca(e.target.value));

        initModal('.modal-overlay');
        renderCatalogo();
    });

    initSeEsiste('.accordion', () => inizializzaAccordion());
});

Dropdown — 3 varianti

A) Bottoni filtro (consigliato) — zero edge case, stato visibile

<div class="filtri-bottoni" data-filtri="categoria"></div>

B) Tendina UL/LI — più compatta, pattern verifica-2304

<div class="dropdown-custom" data-dropdown="categoria" tabindex="0">
    <div class="dropdown-label">Categoria: <span class="dropdown-valore">Tutte</span></div>
    <ul class="dropdown-opzioni"></ul>
</div>

C) <select> nativo — fallback rapido se il tempo scarseggia

<select class="dropdown" onchange="filtraCategoria(this.value)">
    <option value="tutte">Tutte</option>
</select>

Dettaglio in-pagina (alternativa al modal)

Pattern verifica-2304/2101: niente modal, il dettaglio sostituisce la griglia.

<section class="pane"></section>
function apriDettaglio(prodotto) {
    mostraDettaglio('.pane', prodotto, {
        campi:      { titolo:'name', testo:'description',
                      prezzo:'price', categoria:'category' },
        campiExtra: [{ etichetta:'Giacenza', campo:'stock' },
                     { etichetta:'Arrivo',   campo:'arrivedAt' }],
        onIndietro: () => renderCatalogo(),
        scrollInto: true
    });
}
